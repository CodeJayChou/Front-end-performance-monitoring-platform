1. 第一优先级：Middleware 三段实现（核心补全）
1.1 Normalize Middleware（STRUCTURAL）
export const normalizeMiddleware: Middleware = {
  name: "normalize",
  stage: "STRUCTURAL",
  priority: 100,

  run(event) {
    if (!event.timestamp) event.timestamp = Date.now();

    if (!event.platform) event.platform = "web";

    if (!event.id) event.id = crypto.randomUUID();

    if (!event.context) event.context = {};

    return event;
  },
};
1.2 Context Middleware（CONTEXTUAL）

⚠️ 注意：这里只允许“读 scope”，不能再写 hub 注入逻辑

export const contextMiddleware: Middleware = {
  name: "context",
  stage: "CONTEXTUAL",
  priority: 100,

  run(event, ctx) {
    // ctx = middleware runtime context（来自 client）
    if (ctx?.breadcrumbs) {
      event.context.breadcrumbs = ctx.breadcrumbs;
    }

    if (ctx?.user) {
      event.context.user = ctx.user;
    }

    return event;
  },
};
1.3 Policy Middleware（过滤 + 采样）
export const policyMiddleware: Middleware = {
  name: "policy",
  stage: "POLICY",
  priority: 100,

  run(event, ctx) {
    // filter
    if (ctx?.disabledTypes?.includes(event.type)) {
      return null;
    }

    // sampling
    const rate = ctx?.sampleRate ?? 1;
    if (Math.random() > rate) return null;

    return event;
  },
};
2. 第二优先级：Scope（运行时上下文生命周期）

这是你当前体系最缺的一块

2.1 Scope 核心结构
export class Scope {
  private stack: Record<string, any>[] = [];

  push(layer: Record<string, any>) {
    this.stack.push(layer);
  }

  pop() {
    this.stack.pop();
  }

  getContext() {
    return Object.assign({}, ...this.stack);
  }
}
2.2 Hub（补完整）
export class Hub {
  constructor(private scope: Scope) {}

  applyToEvent(event: BaseEvent) {
    event.context = this.scope.getContext();
  }

  pushScope(layer: Record<string, any>) {
    this.scope.push(layer);
  }

  popScope() {
    this.scope.pop();
  }
}
2.3 使用方式（关键）
// route change
client.getHub().pushScope({
  route: location.pathname,
});

// modal open
client.getHub().pushScope({
  modal: "login",
});
3. 第三优先级：Integration Manager（去手动数组化）
3.1 Registry
export class IntegrationManager {
  private map = new Map<string, Integration>();

  register(i: Integration, client: Client) {
    if (this.map.has(i.name)) return;

    this.map.set(i.name, i);
    i.setup(client);
  }

  remove(name: string) {
    const i = this.map.get(name);
    i?.teardown?.();
    this.map.delete(name);
  }

  clear() {
    this.map.forEach(i => i.teardown?.());
    this.map.clear();
  }
}
3.2 Client 集成
constructor(...) {
  this.integrationManager = new IntegrationManager();
}

registerIntegration(i: Integration) {
  this.integrationManager.register(i, this);
}

4.2 HTTP Transport（批量）
export class HttpTransport implements Transport {
  private buffer: BaseEvent[] = [];

  constructor(private dsn: string) {}

  send(event: BaseEvent) {
    this.buffer.push(event);

    if (this.buffer.length >= 10) {
      this.flush();
    }
  }

  flush() {
    fetch(this.dsn, {
      method: "POST",
      body: JSON.stringify(this.buffer),
    });

    this.buffer = [];
  }
}

export function validateEvent(e: any): e is BaseEvent {
  if (!e) return false;
  if (!e.type) return false;
  if (!e.timestamp) return false;
  if (!e.payload) return false;
  return true;
}
